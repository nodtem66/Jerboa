Install

Put this code at the last line in HEAD tag 

<script type="text/javascript" src="http://github.com/nodtem66/jerboa/raw/master/jerboa.js" >
</script>

<script>
  Jerboa('id of element to be write','http://github.com/nodtem66/jerboa/raw/master/');
  //Optional
  Jerboa.save = function()
  {
    /* write the code to fire when Jerboa ends (Click 'finish')
    Example:
      When Jerboa finished
      Send the HTML code to script 'save.php' with a number of parameters
    */
    var codeHTML = document.getElementById('id of element').innerHTML;
    Jerboa.send({
      url: 'save.php?param_1=value¶m2=value'
      ,method: 'post'
      ,'param_3': 'value'
      ,'param_4': 'value'
      ,'param_5': codeHTML
    });
  }
</script>
Then, set the id of elementin you BODY tag 
<div id="...."></div>
